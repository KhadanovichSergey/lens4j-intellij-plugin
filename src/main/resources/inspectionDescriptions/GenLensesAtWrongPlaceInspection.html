This inspection reports when the @GenLenses annotation is used at wrong place.
@GenLenses annotation can be used only with top-level classes. Interfaces and inner classes are not allowed. 


    // incorrect
    @GenLenses(lenses = @Lens(path = "name"))
    interface Fly {
        String getName();
    }

    // incorrect
    class Outer {
        @GenLenses(lenses = @Lens(path = "name"))
        class Inner {
            String name;

            // accessors
        }
    }

    // incorrect
    @GenLenses(lenses = @Lens(path = "value"))
    class Outer<A> {
        A value;
    }



    // correct
    @GenLenses(lenses = @Lens(path = "name"))
    class Bird {
        String name;
    }
