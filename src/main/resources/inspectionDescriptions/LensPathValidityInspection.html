This inspection reports when the @Lens#path is not valid.
@Lens#path should not be blank and all properties should be found in the corresponding classes 


    // incorrect
    @GenLenses(lenses = @Lens(path = ""))
    class Payment {
        String name;
    }

    // incorrect
    @GenLenses(lenses = @Lens(path = "wrongName"))
    class Payment {
        String name;
    }



    // correct
    @GenLenses(lenses = @Lens(path = "name"))
    class Payment {
        String name;
    }
